C.1 Atomic Wallet Debit (Escrow Checkout)

// lib/actions/rental-checkout.ts (excerpt)
export async function createRentalOrder(data: CheckoutInput) {
  const session = await auth();
  if (!session?.user?.id) return { error: "Not authenticated" };

  const totalPaid = rentalFee + platformFee + securityDeposit;

  // Atomic: check balance AND debit in one transaction
  const [updatedUser, order] = await prisma.$transaction([
    prisma.user.update({
      where: { id: session.user.id, walletBalance: { gte: totalPaid } },
      data:  { walletBalance: { decrement: totalPaid } },
    }),
    prisma.rentalOrder.create({ data: { ...orderData } }),
  ]);
  // If walletBalance < totalPaid, the where-clause fails and the
  // entire transaction is rolled back — no partial state possible.
}

C.2 Rental Lifecycle Status Transition (Approval)

// lib/actions/rental-transitions.ts (excerpt)
export async function approveRentalRequest(orderId: string) {
  const session = await auth();
  const order   = await prisma.rentalOrder.findUnique({ where: { id: orderId } });

  if (order?.ownerId !== session?.user?.id) return { error: "Not authorized" };
  if (order.status !== "REQUESTED")         return { error: "Invalid status" };

  const newHistory = [
    ...(order.statusHistory as any[]),
    { status: "APPROVED", changedAt: new Date().toISOString(), note: "Owner approved" }
  ];

  await prisma.rentalOrder.update({
    where: { id: orderId },
    data:  { status: "APPROVED", statusHistory: newHistory },
  });

  // Notify renter
  await prisma.notification.create({
    data: { userId: order.renterId, type: "RENTAL_UPDATE",
            message: `เจ้าของอนุมัติคำขอเช่าของคุณแล้ว — ${order.refCode}`,
            link: `/rental/orders/${orderId}` }
  });
  return { success: true };
}

C.3 Digital Handshake Confirmation (Pickup)

// lib/actions/rental-transitions.ts (excerpt)
export async function confirmRentalPickup(
  orderId:       string,
  photos:        string[],
  conditionNote: string
) {
  const session = await auth();
  const order   = await prisma.rentalOrder.findUnique({ where: { id: orderId } });
  const isRenter = order?.renterId === session?.user?.id;
  const isOwner  = order?.ownerId  === session?.user?.id;

  const updateData = isRenter
    ? { pickupRenterConfirm: true, pickupRenterPhotos: photos }
    : { pickupOwnerConfirm:  true, pickupOwnerPhotos:  photos };

  const updated = await prisma.rentalOrder.update({
    where: { id: orderId },
    data:  updateData,
  });

  // Both confirmed? → transition to ACTIVE
  if (updated.pickupRenterConfirm && updated.pickupOwnerConfirm) {
    await prisma.rentalOrder.update({
      where: { id: orderId },
      data:  { status: "ACTIVE", actualPickupAt: new Date(),
               statusHistory: [..., { status: "ACTIVE", changedAt: new Date() }] }
    });
  }
  return { success: true };
}
